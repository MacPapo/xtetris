531void startCpuGame(player *pg1, player* cpu, tet *piece, int* tetPieces, WINDOW* s_preview, WINDOW* s_score)
532{
533 int pieces = T_NUM * (T_PIECES * 2);
534 int quit = 0;
535 int choice = 0;
536 int countCurrentPiece;
537
538 int bestRow = 0;
539 int zeroInterval = 0;

541 int position_x = 0;
542
543 int *currentScore = &pg1->score;
544 int *cpuScore = &cpu->score;
545
546 tet preview_piece = {piece->tet + 1, 0};
547
548 refreshPreview(s_preview, &preview_piece);
549 refreshGameField(&position_x, piece, pg1);
550 refreshCpuScore(tetPieces[piece->tet], currentScore, cpuScore, s_score);
551
552 do
553 {
554 keypad(s_preview, TRUE);
555 choice = wgetch(s_preview);
556 colorField(pg1);
557
558 countCurrentPiece = tetPieces[piece->tet];
559 if (countCurrentPiece == 0)
560 changePiece(s_score);
561
562 switch (choice)
563 {
564
565 case 'n':
566 nextPiece(piece, &preview_piece);
567 refreshCpuScore(tetPieces[piece->tet], currentScore, cpuScore, s_score);
568 break;
569
570 /* case: back piece */
571 case 'b':
572 backPiece(piece, &preview_piece);
573 refreshCpuScore(tetPieces[piece->tet], currentScore, cpuScore, s_score);
574 break;
575
576 /* case: rotate piece */
577 case 'r':
578 rotatingPiece(piece);
579 break;
580
581 case KEY_LEFT:
582 position_x -= 2;
583 break;
584
585 case KEY_RIGHT:
586 position_x += 2;
587 break;
588
589 case KEY_DOWN:
590 if (countCurrentPiece)
591 {
592 fallingPiece(pg1);
593 tetPieces[piece->tet] -= 1;
594 pieces -= 1;
595 bigBrain(cpu, &bestRow, &zeroInterval, tetPieces);
596 *currentScore += checkAndReverseRows(pg1, cpu);
597 *cpuScore += checkAndReverseRows(cpu, pg1);
598 wrefresh(cpu->window);
599 refreshCpuScore(tetPieces[piece->tet], currentScore, cpuScore, s_score);
600 }
601 break;
602
603 case 'q':
604 quit = 1;
605 break;
606
607 default:
608 break;
609 }
610
611 if (checkGameOver(pg1->gameField))
612 {
613 initGameOver(pg1->score);
614 return;
615 } else if (checkGameOver(cpu->gameField)) {
616 initGameOver(cpu->score);
617 return ;
618 }
619
620 if (pieces == 0 || quit)
621 {
622 initGameOver(pg1->score);
623 return ;
624 }
625
626 refreshPreview(s_preview, &preview_piece);
627 refreshGameField(&position_x, piece, pg1);
628
629 } while (pieces&& !quit);
630
631 return ;
632}
