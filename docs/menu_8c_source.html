48{
49 /* declaring windows */
50 WINDOW *w_menu = NULL, *w_title = NULL, *w_cmds = NULL, *w_desc = NULL;
51 WINDOW *s_menu, *s_desc;

57
58 /* initialize modCounter & userChoice */

61 /* defining parameter for creating menu box */
62 menuY = HCENTER - (MENU_TITLE_Y) + 1;
63 menuX = WCENTER + 1;
64 /* defining parameters for creating description box*/
65 descY = HCENTER - (MENU_TITLE_Y) + 1;

68 /* background color & initialize the menu */
69 bkgd(COLOR_PAIR(6));
81 initMenu(w_menu, w_title, w_cmds, w_desc);
82
83 /* creating the 2 little box that contain the text */
84 s_desc = newwin(DESC_H - 2, DESC_W - 2, descY, descX);
85 wbkgd(s_desc, COLOR_PAIR(3));
86 s_menu = newwin(MENU_H - 2, DESC_W - 2, menuY, menuX);
87 wbkgd(s_menu, COLOR_PAIR(3));
89 /* defining key input & echo disable */
90 keypad(s_menu, TRUE);
91 noecho();
92 curs_set(0);
93
94 do
95 {
105 highlightMenu(s_menu, s_desc, modCounter);
106 user_choice = wgetch(s_menu);
107 switch (user_choice)
108 {
109 case KEY_UP:
110 modCounter--;
111 highlightMenu(s_menu, s_desc, modCounter);
112 if(modCounter == -1)
113 {
114 modCounter = 4;
115 halfdelay(1);
116 }
117 break;
118
119 case KEY_DOWN:
120 modCounter++;
121 highlightMenu(s_menu, s_desc, modCounter);
122 if(modCounter == 5)
123 {
124 modCounter = 0;
125 halfdelay(1);
126 }
127 break;
128
129 default:
130 highlightMenu(s_menu, s_desc, modCounter);
131 break;
132 }
133 halfdelay(1);
134 if(user_choice == 10 && modCounter != 3)
135 {
142 return modCounter;
143 }
144 }
145 while(user_choice != 'q');
146
147 return 4;
148}
