142int startTheGame(player* pg1, player* pg2, tet* piece, int* tetPieces, int* pieces, int* turn, WINDOW* preview, WINDOW* score)
143{
144 int choice = 0;
145 int action = 0;
146
147 int positionX = 0;
148 tet preview_piece = {piece->tet + 1, 0};
149
150 int countCurrentPiece = 0;
151 int *currentScore = &pg1->score;
152
153 if (*pieces == 0)
154 return 1;
155
156 refreshPreview(preview, &preview_piece);
157 refreshGameField(&positionX, piece, pg1);
158 refreshMultiScore(tetPieces[piece->tet], currentScore, pg2->score, turn, score);
160 do
161 {
162 keypad(pg1->window, TRUE);
170 colorField(pg2);
171 choice = wgetch(pg1->window);
172 colorField(pg1);
173
174 countCurrentPiece = tetPieces[piece->tet];
175 if (countCurrentPiece == 0)
176 changePiece(score);
177
178 switch (choice)
179 {
180 case 'n':
189 nextPiece(piece, &preview_piece);
190 refreshMultiScore(tetPieces[piece->tet], currentScore, pg2->score, turn, score);
191 break;
192
193 /* case: back piece */
194 case 'b':
203 backPiece(piece, &preview_piece);
204 refreshMultiScore(tetPieces[piece->tet], currentScore, pg2->score, turn, score);
205 break;
206
207 /* case: rotate piece */
208 case 'r':
215 rotatingPiece(piece);
216 break;
217
218 case KEY_LEFT:
219 positionX -= 2;
220 break;
221
222 case KEY_RIGHT:
223 positionX += 2;
224 break;
225
226 case KEY_DOWN:
227 if (countCurrentPiece)
228 {
236 fallingPiece(pg1);
237 tetPieces[piece->tet] -= 1;
238 *pieces -= 1;
239 *currentScore += checkAndReverseRows(pg1, pg2);
240 refreshMultiScore(tetPieces[piece->tet], currentScore, pg2->score, turn, score);
241 action = 1;
242 }
243 break;
244
245 case 'q':
246 initMultiQuit();
247 return 0;
248 break;
249
250 case 'h':
251 break;
252
253 default:
254 break;
255 }
256
257 refreshPreview(preview, &preview_piece);
258 refreshGameField(&positionX, piece, pg1);
259
260 } while (!action);
261
262 if (checkGameOver(pg1->gameField))
263 {
264 multiGameOver(pg2->score, turn);
265 return 0;
266 }
267
268 colorField(pg1);
283 return startTheGame(pg2, pg1, piece, tetPieces, pieces, changeTurn(turn), preview, score);
